import { useRef, useEffect, useState } from "react";
import useIsInViewport from "./hooks/useIsInViewport";
import PlayIcon from './assets/icons/mixtape.svg';

const Descubre = ({
  index,
  author,
  videoURL,
  authorLink,
  lastVideoIndex,
  miniatura,
}) => {
  const video = useRef();
  const isInViewport = useIsInViewport(video);
  const [isPaused, setIsPaused] = useState(true);

  useEffect(() => {
    const currentVideo = video.current;

    const playWhenReady = () => {
      if (isInViewport) {
        currentVideo.play();
        setIsPaused(false);
      } else {
        currentVideo.pause();
        setIsPaused(true);
      }
    };

    currentVideo.addEventListener('canplay', playWhenReady);

    return () => {
      currentVideo.removeEventListener('canplay', playWhenReady);
    };
  }, [isInViewport]);

  useEffect(() => {
    return () => {
      const currentVideo = video.current;
      if (currentVideo) {
        currentVideo.pause();
        setIsPaused(true);
      }
    };
  }, []);
  

  const togglePlayPause = () => {
    if (video.current.paused) {
      video.current.play();
      setIsPaused(false);
    } else {
      video.current.pause();
      setIsPaused(true);
    }
  };

  return (
    <div className="slider-children">
        <video
          className="video"
          ref={video}
          id={index}
          autoPlay={index === 1}
          poster={miniatura}
          loop
          playsInline
          >
          <source src={videoURL} type="video/mp4" />
        </video>

        <div className="video-overlay" onClick={togglePlayPause}>
            {isPaused && <img src={PlayIcon} alt="Play" className="play-icon"/>}
        </div>
        {/* <div className="video-overlay" onClick={togglePlayPause}>
          {isPaused && <i className="fas fa-play"></i>}
        </div> */}

        <div className="video-content">
          <p className="m-0">@{author}</p>
          <p>
            Post by <a href={authorLink}>{author} </a> en Mixtape
          </p>
        </div>
    </div>

  );
};

export default Descubre;
